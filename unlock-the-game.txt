<script>
// SOUND EFFECTS
const flipSound = new Audio("https://assets.mixkit.co/sfx/preview/mixkit-card-flip-1563.mp3");
const correctSound = new Audio("https://assets.mixkit.co/sfx/preview/mixkit-correct-answer-tone-2870.mp3");
const wrongSound = new Audio("https://assets.mixkit.co/sfx/preview/mixkit-wrong-answer-fail-notification-946.mp3");
const winSound = new Audio("https://assets.mixkit.co/sfx/preview/mixkit-winning-chimes-2015.mp3");

const team = [
  { first: "Marco", last: "Nimani", clues: ["I am 25 🎂","I like soccer ⚽","I make music 🎵"] },
  { first: "Jurdian", last: "Thielman", clues: ["Speaks 5 languages 🌍","Loves to travel ✈️","Studies Finance 📊"] },
  { first: "Katja", last: "Prysiazhniuk", clues: ["I have a cat named Dmitri 🐱","I can do origami 🧩","I love traveling 🌏"] }
];

let cards, firstPick, secondPick, matches, moves, timer, interval;

function initGame() {
  cards = [];
  firstPick = null;
  secondPick = null;
  matches = 0;
  moves = 0;
  timer = 0;

  document.getElementById("moves").innerText = moves;
  document.getElementById("timer").innerText = timer;
  document.getElementById("board").innerHTML = "";
  document.getElementById("clue-box").innerHTML = "";
  document.getElementById("final-screen").style.display = "none";
  document.getElementById("team-display").innerHTML = "";

  clearInterval(interval);
  interval = setInterval(() => {
    timer++;
    document.getElementById("timer").innerText = timer;
  }, 1000);

  team.forEach(m => {
    cards.push({ type: "first", value: m.first, id: m.first });
    cards.push({ type: "last", value: m.last, id: m.first });
  });

  cards.sort(() => Math.random() - 0.5);

  const board = document.getElementById("board");

  cards.forEach(card => {
    const div = document.createElement("div");
    div.classList.add("card");

    div.innerHTML = `
      <div class="card-inner">
        <div class="card-front">?</div>
        <div class="card-back">${card.value}</div>
      </div>
    `;

    div.dataset.id = card.id;
    div.dataset.type = card.type;

    div.onclick = () => handleClick(div);
    board.appendChild(div);
  });
}

function handleClick(card) {
  if (card.classList.contains("flip") || card.classList.contains("matched") || secondPick) return;

  flipSound.play();
  card.classList.add("flip");

  if (!firstPick) {
    firstPick = card;
  } else {
    secondPick = card;
    moves++;
    document.getElementById("moves").innerText = moves;
    checkMatch();
  }
}

function checkMatch() {
  if (
    firstPick.dataset.id === secondPick.dataset.id &&
    firstPick.dataset.type !== secondPick.dataset.type
  ) {
    correctSound.play();

    firstPick.classList.add("matched");
    secondPick.classList.add("matched");

    const member = team.find(m => m.first === firstPick.dataset.id);

    document.getElementById("clue-box").innerHTML =
      `<h3>${member.first} ${member.last}</h3>` +
      `<ul>${member.clues.map(c => `<li>${c}</li>`).join("")}</ul>`;

    matches++;

    if (matches === team.length) {
      clearInterval(interval);
      winSound.play();
      showFinal();
    }

    resetPick();
  } else {
    wrongSound.play();
    setTimeout(() => {
      firstPick.classList.remove("flip");
      secondPick.classList.remove("flip");
      resetPick();
    }, 700);
  }
}

function resetPick() {
  firstPick = null;
  secondPick = null;
}

function showFinal() {
  document.getElementById("final-screen").style.display = "block";
  const container = document.getElementById("team-display");

  team.forEach(m => {
    const card = document.createElement("div");
    card.classList.add("team-card");

    card.innerHTML =
      `<strong>${m.first} ${m.last}</strong>` +
      `<ul>${m.clues.map(c => `<li>${c}</li>`).join("")}</ul>`;

    container.appendChild(card);
  });
}

function resetGame() {
  initGame();
}

initGame();
</script>
